
#include <gtest/gtest.h>

#include <SomePlatformer/GameProgressAndStateInfo.hpp>


TEST(SomePlatformer_GameProgressAndStateInfoTest, can_be_created_without_blowing_up)
{
   SomePlatformer::GameProgressAndStateInfo game_progress_and_state_info;
}


TEST(SomePlatformer_GameProgressAndStateInfoTest, TYPE__has_the_expected_value)
{
   EXPECT_STREQ(
     "SomePlatformer/GameProgressAndStateInfo",
     SomePlatformer::GameProgressAndStateInfo::TYPE
   );
}


TEST(SomePlatformer_GameProgressAndStateInfoTest, type__has_the_expected_value_matching_TYPE)
{
   SomePlatformer::GameProgressAndStateInfo game_progress_and_state_info;
   EXPECT_EQ(SomePlatformer::GameProgressAndStateInfo::TYPE, game_progress_and_state_info.get_type());
}
